\begin{frame}{KVM}

\end{frame}


\begin{frame}{Funkcje}
  \begin{itemize}
    \item Emulacja pełnej maszyny wraz z BIOS
      \begin{itemize}
        \item SMP
        \item PAE
        \item x86/x86\_64
      \end{itemize}
    \item Emulacja urządzeń
      \begin{itemize}
        \item HDD, CDROM
        \item Karta sieciowa
        \item USB, port szeregowy, równoległy
        \item karta graficzna
        \item karta dźwiękowa
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Funkcje cd}

      \end{itemize}
    \item Migracja online
  \end{itemize}
\end{frame}
